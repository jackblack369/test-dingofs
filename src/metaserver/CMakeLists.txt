add_subdirectory(common)
add_subdirectory(s3)

file(GLOB DINGOFS_METASERVER_SRCS
    "*.cpp"
    "copyset/*.cpp"
    "storage/*.cpp"
    "superpartition/*.cpp"
    "streaming/*.cpp"
)
list(REMOVE_ITEM DINGOFS_METASERVER_SRCS
    "${CMAKE_CURRENT_SOURCE_DIR}/main.cpp"
)

add_library(metaserver_lib 
    ${DINGOFS_METASERVER_SRCS} 
)
target_link_libraries(metaserver_lib
    PROTO_OBJS
    dingofs_common
    metaserver_common
    dingofs_utils
    rpcclient
    dynamic_vlog
    local_fs
    dingofs_threading
    metaserver_s3
    dingofs_base_lib
    braft::braft
    brpc::brpc
    RocksDB::rocksdb
    absl::log_internal_message
    absl::cleanup
    absl::btree
    absl::memory
    absl::utility
)

add_executable(dingo-metaserver main.cpp)
target_link_libraries(dingo-metaserver
    metaserver_lib
)
